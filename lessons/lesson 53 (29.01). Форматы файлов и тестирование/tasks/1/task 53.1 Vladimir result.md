import csv

def calculate_average_grades(filename):
    with open(filename, 'r', encoding='utf-8') as file:
        reader = csv.DictReader(file)
        subjects = reader.fieldnames[1:]  # все столбцы кроме первого (Name)
        sums = {subject: 0 for subject in subjects}
        count = 0
        
        for row in reader:
            for subject in subjects:
                try:
                    sums[subject] += int(row[subject])
                except ValueError:
                    # обработка нечисловых значений, если необходимо
                    pass
            count += 1
        
        if count > 0:
            for subject in subjects:
                average = sums[subject] / count
                print(f"{subject}: {average}")

calculate_average_grades("students.csv")
