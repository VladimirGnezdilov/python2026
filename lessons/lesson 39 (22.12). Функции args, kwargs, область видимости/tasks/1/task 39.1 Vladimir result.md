### 6. Исправленное решение
```python
def mean_abs_deviation(*points):
    if not points:
        raise ValueError("Нужно передать хотя бы один балл")
    avg = sum(points) / len(points)
    deviations = []
    for point in points:
        deviations.append(round(abs(avg - point), 1))
    return tuple(deviations)

# Тесты
print(mean_abs_deviation(3, 5, 7, 9))  # (3.0, 1.0, 1.0, 3.0)
existing_scores = [10, 8, 6]
print(mean_abs_deviation(*existing_scores))  # (2.0, 0.0, 2.0)
```
